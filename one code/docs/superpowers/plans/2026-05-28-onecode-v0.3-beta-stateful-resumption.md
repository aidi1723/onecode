# OneCode v0.3-beta Stateful Resumption Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add explicit `--resume-from` support so a new OneCode run can import verified assets from a named prior run and skip already-ready writes.

**Architecture:** Introduce `resumption.py` for manifest audit and ready-asset indexing, then thread the resulting `ResumeState` through context, runner, checkpoint, and CLI. Keep v0.2 sovereignty behavior intact: all writes still pass `LogosGate.preflight()` and `PathGuard`.

**Tech Stack:** Python 3 standard library, `unittest`, `argparse`, JSON files, SHA256 hashing.

---

## File Structure

- Create `src/onecode/kernel/resumption.py`: `ReadyAsset`, `ResumeState`, and previous-manifest audit helpers.
- Modify `src/onecode/kernel/context.py`: carry optional resume metadata and load resume state at context creation.
- Modify `src/onecode/kernel/runner.py`: skip ready resumed `write_text` assets and write skipped evidence.
- Modify `src/onecode/kernel/checkpoint.py`: include `resumed_from` and ready asset summaries in manifest records.

- Create `tests/test_resumption.py`: manifest audit and ready asset indexing tests.
- Modify `tests/test_context.py`: context resume state tests.
- Modify `tests/test_runner_cli.py`: skip, missing-asset write, and CLI resume tests.
- Modify `tests/test_checkpoint.py`: resumed metadata persistence tests.

```python
from dataclasses import dataclass
from pathlib import Path
from uuid import uuid4

from onecode.kernel.hexagram import BUILD_ENTRY, HexagramStatusCode
from onecode.kernel.resumption import ResumeState, load_resume_state


@dataclass(frozen=True)
class OneCodeContext:
    run_id: str
    workspace_root: Path
    evidence_root: Path
    state: HexagramStatusCode
    turn_index: int
    manifest_path: Path
    http_timeout_seconds: float
    resume_from_run_id: str | None = None
    resume_state: ResumeState | None = None


def create_context(
    workspace_root: Path,
    http_timeout_seconds: float = 60,
    run_id: str | None = None,
    resume_from_run_id: str | None = None,
) -> OneCodeContext:
    if http_timeout_seconds <= 0:
        raise ValueError("http_timeout_seconds must be greater than zero")

    resolved_workspace = workspace_root.resolve()
    resolved_workspace.mkdir(parents=True, exist_ok=True)
    selected_run_id = run_id or uuid4().hex
    evidence_root = resolved_workspace / ".onecode" / "runs" / selected_run_id
    checkpoints_root = evidence_root / "checkpoints"
    checkpoints_root.mkdir(parents=True, exist_ok=True)
    resume_state = load_resume_state(resolved_workspace, resume_from_run_id) if resume_from_run_id else None

    return OneCodeContext(
        run_id=selected_run_id,
        workspace_root=resolved_workspace,
        evidence_root=evidence_root,
        state=BUILD_ENTRY,
        turn_index=0,
        manifest_path=evidence_root / "manifest.json",
        http_timeout_seconds=http_timeout_seconds,
        resume_from_run_id=resume_from_run_id,
        resume_state=resume_state,
    )
```

```python
import json
from dataclasses import dataclass
from pathlib import Path

from onecode.kernel.checkpoint import sha256_file


@dataclass(frozen=True)
class ReadyAsset:
    path: str
    sha256: str
    source_run_id: str
    source_turn_index: int


@dataclass(frozen=True)
class ResumeState:
    source_run_id: str
    ready_assets: dict[str, ReadyAsset]


def load_json(path: Path) -> dict:
    return json.loads(path.read_text(encoding="utf-8"))


def normalize_asset_path(workspace_root: Path, payload_path: str) -> str | None:
    path = Path(payload_path)
    root = workspace_root.resolve()
    target = path if path.is_absolute() else root / path
    try:
        return str(target.resolve().relative_to(root))
    except ValueError:
        return None


def checkpoint_to_ready_asset(workspace_root: Path, source_run_id: str, checkpoint: dict) -> ReadyAsset | None:
    if checkpoint.get("status") != "completed":
        return None
    if checkpoint.get("intent_type") != "write_text":
        return None
    if checkpoint.get("decision") != "allowed":
        return None

    payload = checkpoint.get("payload")
    if not isinstance(payload, dict):
        return None
    payload_path = payload.get("path")
    payload_sha = payload.get("sha256")
    if not isinstance(payload_path, str) or not isinstance(payload_sha, str):
        return None

    relative_path = normalize_asset_path(workspace_root, payload_path)
    if relative_path is None:
        return None

    asset_path = workspace_root.resolve() / relative_path
    if not asset_path.exists():
        return None
    if sha256_file(asset_path) != payload_sha:
        return None

    return ReadyAsset(
        path=relative_path,
        sha256=payload_sha,
        source_run_id=source_run_id,
        source_turn_index=int(checkpoint.get("turn_index", 0)),
    )


def load_resume_state(workspace_root: Path, resume_from_run_id: str) -> ResumeState:
    root = workspace_root.resolve()
    manifest_path = root / ".onecode" / "runs" / resume_from_run_id / "manifest.json"
    if not manifest_path.exists():
        return ResumeState(source_run_id=resume_from_run_id, ready_assets={})

    manifest = load_json(manifest_path)
    ready_assets: dict[str, ReadyAsset] = {}
    for record in manifest.get("checkpoints", []):
        checkpoint_path_value = record.get("path") if isinstance(record, dict) else None
        if not isinstance(checkpoint_path_value, str):
            continue
        checkpoint_path = Path(checkpoint_path_value)
        if not checkpoint_path.exists():
            continue
        checkpoint = load_json(checkpoint_path)
        ready_asset = checkpoint_to_ready_asset(root, resume_from_run_id, checkpoint)
        if ready_asset is not None:
            ready_assets[ready_asset.path] = ready_asset

    return ResumeState(source_run_id=resume_from_run_id, ready_assets=ready_assets)
```
