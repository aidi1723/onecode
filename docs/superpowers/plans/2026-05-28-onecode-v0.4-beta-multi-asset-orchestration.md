# OneCode v0.4-beta Multi-Asset Orchestration Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Extend `onecode run` so one invocation can process multiple `write_text` assets with ordered checkpoints, resume skips, fail-fast halt behavior, and ledger summary counts.

**Architecture:** Keep the existing v0.3 single-asset path compatible by normalizing all run inputs into an ordered list of `ActionIntent` objects. Each intent goes through the existing `LogosGate.preflight()`, resume skip rule, `PathGuard.write_text()`, `write_checkpoint()`, and final `write_ledger()` pipeline. The CLI adds repeatable `--write-text PATH=CONTENT` arguments and rejects mixed old/new write interfaces.

**Tech Stack:** Python 3 standard library, `argparse`, `unittest`, JSON evidence files, existing OneCode kernel modules.

```python
def parse_write_text(value: str) -> tuple[str, str]:
    path, separator, content = value.partition("=")
    if separator == "" or path == "":
        raise ValueError("--write-text must use PATH=CONTENT with a non-empty path")
    return path, content


def build_intents(
    intent_type: str,
    write_path: str | None,
    write_content: str | None,
    command: str | None,
    write_texts: list[str] | None,
) -> list[ActionIntent]:
    if write_texts:
        return [ActionIntent.write_text(path, content) for path, content in map(parse_write_text, write_texts)]
    return [build_intent(intent_type, write_path, write_content, command)]
```

```python
def asset_entry(
    index: int,
    gate_result: dict[str, Any],
    preflight_decision: str,
    intent_type: str,
) -> dict[str, Any]:
    entry = {
        "index": index,
        "status": gate_result["status"],
        "partial": gate_result["partial"],
        "reason": gate_result["reason"],
        "decision": preflight_decision,
        "intent_type": intent_type,
        "payload": gate_result["payload"],
        "resumed": gate_result["status"] == "skipped",
    }
    if "sha256" in gate_result["payload"]:
        entry["sha256"] = gate_result["payload"]["sha256"]
    return entry
```

```python
    intents = build_intents(intent_type, write_path, write_content, command, write_texts)
    assets: list[dict[str, Any]] = []

    for index, intent in enumerate(intents, start=1):
        preflight = gate.preflight(context, intent)
        ready_asset = ready_asset_for_intent(context, intent)

        if preflight.decision == Decision.DENIED:
            gate_result = {
                "status": "denied",
                "partial": False,
                "reason": preflight.reason,
                "payload": preflight.to_dict(),
            }
        elif preflight.decision == Decision.HALTED:
            gate_result = {
                "status": "halted",
                "partial": True,
                "reason": preflight.reason,
                "payload": preflight.to_dict(),
            }
        elif preflight.decision == Decision.ALLOWED and ready_asset is not None:
            gate_result = {
                "status": "skipped",
                "partial": False,
                "reason": "resumed_asset_ready",
                "payload": {
                    "path": ready_asset.path,
                    "sha256": ready_asset.sha256,
                    "source_run_id": ready_asset.source_run_id,
                    "source_turn_index": ready_asset.source_turn_index,
                },
            }
        else:

            def action() -> dict[str, Any]:
                if simulated_action_seconds > 0:
                    time.sleep(simulated_action_seconds)
                if intent.action_type.value == "write_text":
                    return PathGuard.write_text(
                        context.workspace_root,
                        intent.payload["path"],
                        intent.payload["content"],
                    )
                return {
                    "task": task,
                    "run_id": context.run_id,
                    "turn_index": context.turn_index + index,
                }

            gate_result = gate.run_bounded_action(action)

        write_checkpoint(
            context=context,
            payload=gate_result["payload"],
            next_state=COMPLETE,
            status=gate_result["status"],
            partial=gate_result["partial"],
            reason=gate_result["reason"],
            intent_type=intent.action_type.value,
            decision=preflight.decision.value,
        )
        assets.append(asset_entry(index, gate_result, preflight.decision.value, intent.action_type.value))

        if gate_result["status"] in {"denied", "halted"}:
            break
```
